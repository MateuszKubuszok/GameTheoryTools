348 ResultPtr Strategic2PlayerMixedEquilibriumRoutine::solveProblem(
349  LPProblem* problem,
350  const StrategicDataAccessor& data,
351  const StrategicGamePositionsHelper& positionsHelper
352 ) const {
353  typedef map<Identifier, double> DistributionMap;
354 
355  glp_smcp parm;
356  glp_init_smcp(&parm);
357  parm.msg_lev = GLP_MSG_OFF;
358 
359  if (glp_simplex(problem, &parm) != 0 || glp_get_prim_stat(problem) != GLP_FEAS) {
360  // TODO: create MessageFactory
361  return ResultFactory::getInstance().constResult(Message("Failed to calculate the solution"));
362  }
363 
364  const Player& player1 = *( positionsHelper.getPlayers()->begin()->second );
365  const Identifier& player1Name = *( player1.getName() );
366  const Player& player2 = *( (++positionsHelper.getPlayers()->begin())->second );
367  const Identifier& player2Name = *( player2.getName() );
368 
369  const int b1ColStart = 1;
370  const int b2ColStart = b1ColStart + player1.getStrategiesNumber();
371  const int p1ColStart = b2ColStart + player2.getStrategiesNumber();
372  const int p2ColStart = p1ColStart + player1.getStrategiesNumber();
373 
374  DistributionMap player1Distribution;
375  for (const IdentifierPtr& strategy : *player1.getStrategies()) {
376  Index i = player1.getStrategyOrdinal(*strategy);
377  double probability = glp_get_col_prim(problem, p1ColStart + i);
378 
379  player1Distribution.insert( DistributionMap::value_type(*strategy, probability) );
380  }
381 
382  DistributionMap player2Distribution;
383  for (const IdentifierPtr& strategy : *player2.getStrategies()) {
384  Index i = player2.getStrategyOrdinal(*strategy);
385  double probability = glp_get_col_prim(problem, p2ColStart + i);
386 
387  player2Distribution.insert( DistributionMap::value_type(*strategy, probability) );
388  }
389 
390  double player1Payoff = 0.0;
391  double player2Payoff = 0.0;
392  for (const IdentifierPtr& player1Strategy : *player1.getStrategies()) {
393  for (const IdentifierPtr& player2Strategy : *player2.getStrategies()) {
394  Positions positions;
395  positions.insert( Positions::value_type(player1Name, *player1Strategy) );
396  positions.insert( Positions::value_type(player2Name, *player2Strategy) );
397 
398  player1Payoff += data[positions]->getPayoff(player1Name)->get_d() *
399  player1Distribution[*player1Strategy] *
400  player2Distribution[*player2Strategy];
401  player2Payoff += data[positions]->getPayoff(player2Name)->get_d() *
402  player1Distribution[*player1Strategy] *
403  player2Distribution[*player2Strategy];
404  }
405  }
406 
407  ResultBuilderPtr resultBuilder = ResultFactory::getInstance().buildResult();
408 
409 
410  ResultBuilderPtr player1DistributionResult = ResultFactory::getInstance().buildResult();
411  for (const IdentifierPtr& player1Strategy : *player1.getStrategies()) {
412  const MessagePtr& probability = createMessagePtr(player1Distribution[*player1Strategy]);
413  player1DistributionResult->addResult( player1Strategy, probability );
414  }
415  ResultBuilderPtr player2DistributionResult = ResultFactory::getInstance().buildResult();
416  for (const IdentifierPtr& player2Strategy : *player2.getStrategies()) {
417  const MessagePtr& probability = createMessagePtr(player2Distribution[*player2Strategy]);
418  player2DistributionResult->addResult( player2Strategy, probability );
419  }
420 
421  ResultBuilderPtr mixedStrategiesResult = ResultFactory::getInstance().buildResult();
422  mixedStrategiesResult->addResult(
423  player1.getName(),
424  createMessagePtr( player1DistributionResult->build()->getResult() )
425  );
426  mixedStrategiesResult->addResult(
427  player2.getName(),
428  createMessagePtr( player2DistributionResult->build()->getResult() )
429  );
430  const MessagePtr mixedStrategies = createMessagePtr(mixedStrategiesResult->build()->getResult());
431 
432  static const IdentifierPtr mixedStrategiesName = createIdentifierPtr("Mixed Strategies");
433  resultBuilder->addResult( mixedStrategiesName, mixedStrategies );
434 
435  static const IdentifierPtr payoffName = createIdentifierPtr("Payoff");
436  ResultBuilderPtr payoffResult = ResultFactory::getInstance().buildResult();
437 
438  IdentifiersPtr headers = createIdentifiersPtr();
439  headers->push_back( player1.getName() );
440  headers->push_back( player2.getName() );
441  payoffResult->setHeaders( headers );
442 
443  MessagesPtr payoffs = createMessagesPtr();
444  payoffs->push_back( createMessagePtr(player1Payoff) );
445  payoffs->push_back( createMessagePtr(player2Payoff) );
446  payoffResult->addRecord( payoffName, payoffs );
447 
448  resultBuilder->addResult( payoffName, createMessagePtr(payoffResult->build()->getResult()) );
449 
450  return resultBuilder->build();
451 }
