116 void ExecutionDriver::execute(
117  const InputLocation& location,
118  const Identifier& fileName
119 ) const {
120  scoped_ptr<ifstream> file(new ifstream(fileName));
121  if (!file->good()) {
122  stringstream builder;
123  builder << "Executed file <" << fileName << "> does not exists";
124  string errorMessage(builder.str());
125 
126  showError(location, errorMessage);
127  return;
128  }
129 
130  ExecutionDriver driver(outputStream, errorStream, context);
131  Scanner scanner(file.get());
132  Parser parser(scanner, driver);
133 
134  bool success = parser.parse() == 0;
135  ResultPtr result;
136  if (success) {
137  static IdentifierPtr name = createIdentifierPtr("Result");
138  static MessagePtr value = createIdentifierPtr("File executed without errors");
139  result = ResultFactory::getInstance().buildResult()->addResult(name, value).build();
140  } else {
141  static IdentifierPtr name = createIdentifierPtr("Result");
142  static MessagePtr value = createIdentifierPtr("There were errors during execution of this file");
143  result = ResultFactory::getInstance().buildResult()->addResult(name, value).build();
144  }
145  showResult(result);
146 }
