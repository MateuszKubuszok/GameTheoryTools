132 Message StrategicData::toString() const {
133  ResultBuilderPtr resultBuilder = ResultFactory::getInstance().buildResult();
134 
135  static const IdentifierPtr positionName = createIdentifierPtr("Position");
136  static const IdentifierPtr payoffName = createIdentifierPtr("Payoff");
137  IdentifiersPtr playersNames = createIdentifiersPtr();
138  for (Index i = 0; i < positionsHelper.getPlayers()->size(); i++)
139  playersNames->push_back( createIdentifierPtr(positionsHelper.retrievePlayer(i)) );
140 
141  Index maxPosition = payoffStorage.size();
142  for (Index i = 0; i < maxPosition; i++)
143  if (payoffStorageAllocation.at(i)) {
144  static const IdentifierPtr name = createIdentifierPtr("Value");
145  const PositionsPtr positions = positionsHelper.retrievePositions(i);
146  const NumbersPtr numbers = payoffStorage[i];
147  const IdentifiersPtr strategies = createIdentifiersPtr();
148  const MessagesPtr numbersStr = createMessagesPtr();
149 
150  for (const IdentifierPtr& playerName : (*playersNames)) {
151  strategies->push_back( createIdentifierPtr((*positions)[*playerName]) );
152  numbersStr->push_back( createMessagePtr(
153  (*numbers)[positionsHelper.calculatePlayer(playerName)]
154  ) );
155  }
156 
157  ResultBuilderPtr subresultBuilder = ResultFactory::getInstance().buildResult();
158  subresultBuilder->setHeaders(playersNames)
159  .addRecord(positionName, strategies)
160  .addRecord(payoffName, numbersStr);
161  MessagePtr subresult = createMessagePtr(subresultBuilder->buildRaw()->getResult());
162  resultBuilder->addResult(name, subresult);
163  }
164 
165  return resultBuilder->build()->getResult();
166 }
