45 ResultPtr Strategic2Player0SumMixedEquilibriumRoutine::findResultFor(
46  const GamePtr game,
47  const ConditionsPtr
48 ) const {
49  const StrategicDataAccessorPtr data = dynamic_pointer_cast<StrategicDataAccessor>(game->getData());
50  const Players& players = *data->getPlayers();

52  static Identifier strategicGameName("0-sum Strategic Game with 2 players");

71 LPProblemPtr Strategic2Player0SumMixedEquilibriumRoutine::initializeProblem(
72  const Players& players
73 ) const {
74  const Player& player1 = *( players.begin()->second );
75  const Player& player2 = *( (++players.begin())->second );
76 
77  LPProblemPtr problemPtr(glp_create_prob(), glp_delete_prob);
78  LPProblem* problem(problemPtr.get());
79 
80  glp_set_prob_name( problem, "2 player 0-sum strategic form game" );
81  glp_set_obj_dir( problem, GLP_MAX );
82 
83  glp_add_cols(problem, player2.getStrategiesNumber());
84  for (const IdentifierPtr& player2Strategy : *player2.getStrategies()) {
85  int j = player2.getStrategyOrdinal(*player2Strategy) + 1;
86  glp_set_col_name(problem, j, player2Strategy->c_str());
87  glp_set_col_bnds(problem, j, GLP_LO, 0.0, 0.0);
88  glp_set_obj_coef(problem, j, 1.0);
89  }
90 
91  glp_add_rows(problem, player1.getStrategiesNumber());
92  for (const IdentifierPtr& player1Strategy : *player1.getStrategies()) {
93  int i = player1.getStrategyOrdinal(*player1Strategy) + 1;
94  glp_set_row_name(problem, i, player1Strategy->c_str());
95  glp_set_row_bnds(problem, i, GLP_UP, 0.0, 1.0);
96  }
97 
98  return problemPtr;
99 }

101 void Strategic2Player0SumMixedEquilibriumRoutine::fillUpProblem(
102  LPProblem* problem,
103  const StrategicDataAccessor& data,
104  const StrategicGamePositionsHelper& positionsHelper
105 ) const {
106  const Player& player1 = *( positionsHelper.getPlayers()->begin()->second );
107  const Player& player2 = *( (++positionsHelper.getPlayers()->begin())->second );
108  const Identifier& player1Name = *( player1.getName() );
109  const Identifier& player2Name = *( player2.getName() );
110  const Index upperBound = positionsHelper.getUpperBound();
111 
112  vector<int> player1Strategies(upperBound+1);
113  vector<int> player2Strategies(upperBound+1);
114  vector<double> payoff(upperBound+1);
115 
116  for (Index k = 0; k < upperBound; k++) {
117  const PositionsPtr positions = positionsHelper.retrievePositions(k);
118  const Identifier& player1Strategy = positions->at(player1Name);
119  const Identifier& player2Strategy = positions->at(player2Name);
120  // change 0-indexing used in Model to 1-indexing used in GLPK
121  const Index player1Ordinal = player1.getStrategyOrdinal(player1Strategy) + 1;
122  const Index player2Ordinal = player2.getStrategyOrdinal(player2Strategy) + 1;
123  const double player1Payoff = data[positions]->getPayoff(player1Name)->get_d();
124 
125  // GLPK uses 1 indeksing also in data arrays
126  player1Strategies[k+1] = player1Ordinal;
127  player2Strategies[k+1] = player2Ordinal;
128  payoff[k+1] = player1Payoff; // figure out GMP with GLPK
129  }
130 
131  // vectors uses contignous memory - &x[0] "transforms" it into a C-style array
132  glp_load_matrix(problem, upperBound, &player1Strategies[0], &player2Strategies[0], &payoff[0]);
133 }

135 ResultPtr Strategic2Player0SumMixedEquilibriumRoutine::solveProblem(
136  LPProblem* problem,
137  const StrategicDataAccessor& data,
138  const StrategicGamePositionsHelper& positionsHelper
139 ) const {
140  typedef map<Identifier, double> DistributionMap;
141 
142  glp_smcp parm;
143  glp_init_smcp(&parm);
144  parm.msg_lev = GLP_MSG_OFF;
145 
146  if (glp_simplex(problem, &parm) != 0 ||
147  glp_get_prim_stat(problem) != GLP_FEAS || glp_get_dual_stat(problem) != GLP_FEAS) {
148  // TODO: create MessageFactory
149  return ResultFactory::getInstance().constResult(Message("Failed to calculate the solution"));
150  }
151 
152  double distributionFactor = glp_get_obj_val(problem);
153 
154  const Player& player1 = *( positionsHelper.getPlayers()->begin()->second );
155  const Identifier& player1Name = *( player1.getName() );
156  DistributionMap player1Distribution;
157  for (const IdentifierPtr& strategy : *player1.getStrategies()) {
158  // change 0-indexing used in Model to 1-indexing used in GLPK
159  Index strategyOrdinal = player1.getStrategyOrdinal(*strategy) + 1;
160  double probability = glp_get_row_dual(problem, strategyOrdinal) / distributionFactor;
161 
162  player1Distribution.insert( DistributionMap::value_type(*strategy, probability) );
163  }
164 
165  const Player& player2 = *( (++positionsHelper.getPlayers()->begin())->second );
166  const Identifier& player2Name = *( player2.getName() );
167  DistributionMap player2Distribution;
168  for (const IdentifierPtr& strategy : *player2.getStrategies()) {
169  // change 0-indexing used in Model to 1-indexing used in GLPK
170  Index strategyOrdinal = player2.getStrategyOrdinal(*strategy) + 1;
171  double probability = glp_get_col_prim(problem, strategyOrdinal) / distributionFactor;
172 
173  player2Distribution.insert( DistributionMap::value_type(*strategy, probability) );
174  }
175 
176  double player1Payoff = 0.0;
177  for (const IdentifierPtr& player1Strategy : *player1.getStrategies()) {
178  for (const IdentifierPtr& player2Strategy : *player2.getStrategies()) {
179  Positions positions;
180  positions.insert( Positions::value_type(player1Name, *player1Strategy) );
181  positions.insert( Positions::value_type(player2Name, *player2Strategy) );
182 
183  player1Payoff += data[positions]->getPayoff(player1Name)->get_d() *
184  player1Distribution[*player1Strategy] *
185  player2Distribution[*player2Strategy];
186  }
187  }
188  double player2Payoff = -player1Payoff;
189 
190  ResultBuilderPtr resultBuilder = ResultFactory::getInstance().buildResult();
191 
192 
193  ResultBuilderPtr player1DistributionResult = ResultFactory::getInstance().buildResult();
194  for (const IdentifierPtr& player1Strategy : *player1.getStrategies()) {
195  const MessagePtr& probability = createMessagePtr(player1Distribution[*player1Strategy]);
196  player1DistributionResult->addResult( player1Strategy, probability );
197  }
198  ResultBuilderPtr player2DistributionResult = ResultFactory::getInstance().buildResult();
199  for (const IdentifierPtr& player2Strategy : *player2.getStrategies()) {
200  const MessagePtr& probability = createMessagePtr(player2Distribution[*player2Strategy]);
201  player2DistributionResult->addResult( player2Strategy, probability );
202  }
203 
204  ResultBuilderPtr mixedStrategiesResult = ResultFactory::getInstance().buildResult();
205  mixedStrategiesResult->addResult(
206  player1.getName(),
207  createMessagePtr( player1DistributionResult->build()->getResult() )
208  );
209  mixedStrategiesResult->addResult(
210  player2.getName(),
211  createMessagePtr( player2DistributionResult->build()->getResult() )
212  );
213  const MessagePtr mixedStrategies = createMessagePtr(mixedStrategiesResult->build()->getResult());
214 
215  static const IdentifierPtr mixedStrategiesName = createIdentifierPtr("Mixed Strategies");
216  resultBuilder->addResult( mixedStrategiesName, mixedStrategies );
217 
218  static const IdentifierPtr payoffName = createIdentifierPtr("Payoff");
219  ResultBuilderPtr payoffResult = ResultFactory::getInstance().buildResult();
220 
221  IdentifiersPtr headers = createIdentifiersPtr();
222  headers->push_back( player1.getName() );
223  headers->push_back( player2.getName() );
224  payoffResult->setHeaders( headers );
225 
226  MessagesPtr payoffs = createMessagesPtr();
227  payoffs->push_back( createMessagePtr(player1Payoff) );
228  payoffs->push_back( createMessagePtr(player2Payoff) );
229  payoffResult->addRecord( payoffName, payoffs );
230 
231  resultBuilder->addResult( payoffName, createMessagePtr(payoffResult->build()->getResult()) );
232 
233  return resultBuilder->build();
234 }
