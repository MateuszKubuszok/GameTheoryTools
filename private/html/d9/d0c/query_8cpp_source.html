42  const Context& context

47  createMessagePtr("Property not found in Param nor referred Object");
48 
49  ResultBuilderPtr queryResult = ResultFactory::getInstance().buildResult();
50 
51  Index objectNumber = 0;
52  for (ObjectPtr& objectPtr : *objects) {
53  const Object& object = *objectPtr;
54  const IdentifierPtr objectName = createIdentifierPtr(++objectNumber);
55 
56  ResultBuilderPtr objectResult = ResultFactory::getInstance().buildResult();
57  for (const IdentifierPtr& propertyName : *propertiesNames) {
58  Identifier& property = *propertyName;
59 
60  if (object.respondsTo(property)) {
61  // Object has this property
62 
63  try {
64  MessagePtr propertyValue = createMessagePtr(
65  object.findPropertyWithConditions(context, property, *conditions)->getResult()
66  );
67  objectResult->addResult(propertyName, propertyValue);
68  } catch (const std::exception& e) {
69  MessagePtr errorMessage = createMessagePtr(e.what());
70  objectResult->addResult(propertyName, errorMessage);
71  }
72  } else {
73  const Param& param = object.toParam();
74  if (param) {
75  // Object is a Param - property might belong to referred Object
76 
77  try {
78  const ObjectPtr referredObject = param.getObject(context);
79  if (referredObject->respondsTo(property)) {
80  // Param refers to Object (not a Number or cyclic reference or udefined Param)
81 
82  MessagePtr propertyValue = createMessagePtr(
83  referredObject->findPropertyWithConditions(context, property, *conditions)
84  ->getResult()
85  );
86  objectResult->addResult(propertyName, propertyValue);
87  } else {
88  // Referred Object has no sought property
89 
90  objectResult->addResult(propertyName, paramPropertyNotFound);
91  }
92  } catch (const std::exception& e) {
93  MessagePtr errorMessage = createMessagePtr(e.what());
94  objectResult->addResult(propertyName, errorMessage);
95  }
96  } else {
97  // Object has no sought property whatsoever
98 
99  objectResult->addResult(propertyName, objectPropertyNotFound);
100  }
101  }
102  }
103 
104  MessagePtr identifierValue = createMessagePtr(objectResult->build()->getResult());
105  queryResult->addResult(objectName, identifierValue);
106  }
107 
108  return queryResult->build();
