43 const NumberPtr& PlainDataPiece::getPayoff(
44  const Identifier& playerName
45 ) const {
46  if (!positionsHelper.checkPlayer(playerName))
47  throw ExceptionFactory::getInstance().invalidPlayer(playerName);
48  const size_t positionInStorage = positionsHelper.calculatePlayer(playerName);
49 
50  if (positionInStorage < numbers->size())
51  return (*numbers)[positionInStorage];
52 
53  if (positionsHelper.getPlayers()->size() == 2 && numbers->size() == 1) {
54  // In a matrix game we are allowed to deduce that: (player 2's payoff) = -1 * (player 1's payoff)
55  Numbers& payoffs = const_cast<Numbers&>(*numbers);
56  const Number player1Payoff = *(*numbers)[0];
57  NumberPtr player2Payoff(new Number(player1Payoff * -1.0));
58  payoffs.push_back(player2Payoff);
59 
60  return payoffs[positionInStorage];
61  }
62 
63  throw ExceptionFactory::getInstance()
64  .noParamsForPositions(positionInStorage, positionsHelper.getUpperBound());
65 }

67 bool PlainDataPiece::is0Sum() const {
68  if (positionsHelper.getPlayers()->size() == 2 && numbers->size() == 1)
69  return true;
70 
71  double sum = 0.0;
72  for (NumberPtr& number : *numbers)
73  sum += number->get_d();
74 
75  double delta = 10e-16;
76  return (sum*sum) < delta;
77 }
